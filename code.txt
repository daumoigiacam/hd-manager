// --- UTILS & SHARED LOGIC ---

const toTitleCase = (str) => {
  if (!str) return '';
  return str.toLowerCase().split(' ').map(word => word.charAt(0).toUpperCase() + word.slice(1)).join(' ');
};

const capitalizeFirst = (str) => {
  if (!str) return '';
  return str.charAt(0).toUpperCase() + str.slice(1);
};

const formatInputCurrency = (val) => {
  if (val === '' || val === undefined || val === null) return '';
  const num = parseInt(val.toString().replace(/\D/g, ''), 10);
  return isNaN(num) ? '' : new Intl.NumberFormat('vi-VN').format(num);
};

const parseInputCurrency = (val) => {
  if (val === '' || val === undefined || val === null) return '';
  const num = parseInt(val.toString().replace(/\D/g, ''), 10);
  return isNaN(num) ? '' : num;
};

const formatCurrency = (amount) => {
  if (amount === undefined || amount === null || isNaN(amount)) return '0';
  return new Intl.NumberFormat('vi-VN').format(amount);
};

const formatTime = (dateObj) => {
  if (!dateObj) return '--:--';
  if (typeof dateObj === 'string') return dateObj.substring(11, 16); 
  return dateObj.toLocaleTimeString('vi-VN', { hour: '2-digit', minute: '2-digit' });
};

const calculateTieredBonus = (revenue, target) => {
  if (!target || revenue <= target) return 0;
  const excess = revenue - target;
  if (excess <= 20000000) return excess * 0.02;
  if (excess <= 50000000) return (20000000 * 0.02) + ((excess - 20000000) * 0.05);
  return (20000000 * 0.02) + (30000000 * 0.05) + ((excess - 50000000) * 0.1);
};

const getCustomerDebt = (customerId, allOrders, allPayments) => {
  const totalO = allOrders.filter(o => o.customerId === customerId && !o.isArchived).reduce((s, o) => s + (o.amount||0), 0);
  const totalP = allPayments.filter(p => p.customerId === customerId && !p.isArchived).reduce((s, p) => s + (p.amount||0), 0);
  return totalO - totalP;
};

const getCustomerPoints = (customerId, allOrders) => {
  const totalO = allOrders.filter(o => o.customerId === customerId && !o.isArchived).reduce((s, o) => s + (o.amount||0), 0);
  return Math.floor(totalO / 100000); 
};

const calculateMonthRevenue = (empId, orders) => {
  const currentMonth = getTodayString().substring(0, 7);
  return orders.filter(o => o.empId === empId && o.date.startsWith(currentMonth) && !o.isArchived).reduce((sum, o) => sum + (o.amount||0), 0);
};

const calculateSalaryDetails = (empId, employees, attendance, financials, performance, allCustomers = [], allOrders = [], allPayments = [], holidays = []) => {
  const emp = employees.find(e => e.id === empId);
  if (!emp) return null;
  const perf = performance[empId] || { revenue: 0, overtime: 0 };
  
  let workDaysProbation = 0;
  let workDaysOfficial = 0;
  let holidayBonus = 0; 
  
  const dailyBaseWage = (emp.basicSalary || 0) / STANDARD_WORK_DAYS;
  const probationRate = (emp.probationRate !== undefined ? emp.probationRate : 100) / 100;

  // Tính ngày hết hạn thử việc
  let probationEndDate = null;
  if (emp.startDate && emp.probationDuration > 0) {
    probationEndDate = new Date(emp.startDate);
    probationEndDate.setHours(23, 59, 59, 999);
    if (emp.probationUnit === 'days') probationEndDate.setDate(probationEndDate.getDate() + parseInt(emp.probationDuration));
    else if (emp.probationUnit === 'weeks') probationEndDate.setDate(probationEndDate.getDate() + parseInt(emp.probationDuration) * 7);
    else if (emp.probationUnit === 'months') probationEndDate.setMonth(probationEndDate.getMonth() + parseInt(emp.probationDuration));
  }

  // Quét qua các ngày đi làm để phân tách Lương chính thức vs Lương thử việc
  Object.keys(attendance).forEach(key => {
    if (key.endsWith(`_${empId}`)) {
      const rec = attendance[key];
      const dateStr = key.split('_')[0]; 
      
      if (rec.status === 'present' || rec.status === 'late') {
        const workDate = new Date(dateStr);
        
        // Phân loại ngày làm thử việc hay chính thức
        if (probationEndDate && workDate <= probationEndDate) {
          workDaysProbation += 1;
        } else {
          workDaysOfficial += 1;
        }

        // Dò xem ngày đi làm này có phải ngày Lễ không
        const isHoliday = holidays.find(h => h.date === dateStr);
        if (isHoliday) {
          if (isHoliday.type === 'fixed') holidayBonus += isHoliday.value;
          else if (isHoliday.type === 'percentage') holidayBonus += dailyBaseWage * (isHoliday.value / 100);
        }
      }
    }
  });

  const workDays = workDaysProbation + workDaysOfficial;
  
  // Tính lương cơ bản theo ngày thử việc + ngày chính thức
  const baseSalaryCalc = (workDaysProbation * dailyBaseWage * probationRate) + (workDaysOfficial * dailyBaseWage);
  const supportSalaryCalc = ((emp.supportSalary || 0) / STANDARD_WORK_DAYS) * workDays;

  // Tính lương kinh nghiệm tự động (+50k/tháng kể từ ngày vào làm)
  let experienceSalaryCalc = 0;
  let expMonths = 0;
  if (emp.startDate) {
    const start = new Date(emp.startDate);
    const now = new Date();
    expMonths = (now.getFullYear() - start.getFullYear()) * 12 + (now.getMonth() - start.getMonth());
    if (expMonths > 0) experienceSalaryCalc = expMonths * 50000; 
  }

  const commission = (perf.revenue || 0) * (emp.commissionRate || 0);
  const tieredBonus = calculateTieredBonus(perf.revenue || 0, emp.targetRevenue || 0);
  const overtimePay = (perf.overtime || 0) * (emp.overtimeRate || 0);

  let totalBonus = 0; let totalPenalty = 0; let totalAdvance = 0;
  financials.filter(f => f.empId === empId && !f.isArchived).forEach(record => {
    if (record.type === 'bonus') totalBonus += record.amount;
    if (record.type === 'penalty') totalPenalty += record.amount;
    if (record.type === 'advance') totalAdvance += record.amount;
  });

  totalBonus += holidayBonus;

  let badDebt = 0;
  if (['Kinh doanh', 'Bán hàng'].includes(emp.position)) {
    const myCustomers = allCustomers.filter(c => c.empId === empId);
    badDebt = myCustomers.reduce((sum, cus) => sum + getCustomerDebt(cus.id, allOrders, allPayments), 0);
    if (badDebt < 0) badDebt = 0; 
  }

  const netSalary = baseSalaryCalc + supportSalaryCalc + experienceSalaryCalc + commission + tieredBonus + overtimePay + totalBonus - totalPenalty - totalAdvance - badDebt;
  
  return { 
    workDays, workDaysProbation, workDaysOfficial, expMonths, perf, baseSalaryCalc, supportSalary: supportSalaryCalc, experienceSalary: experienceSalaryCalc, 
    commission, tieredBonus, overtimePay, totalBonus, holidayBonus, totalPenalty, totalAdvance, badDebt, netSalary 
  };
};

function SmartAIAssistant({ employee, company, activeTab, customers, orders, expenses }) {

  const [isLoading, setIsLoading] = useState(false);

  const handleSend = async (e) => {
    e.preventDefault();
    if (!input.trim() || isLoading) return;

    const newMessages = [...messages, { role: 'user', text: capitalizeFirst(input) }];
    setMessages(newMessages);
    setInput('');
    setIsLoading(true);

    const todaysOrders = orders.filter(o => o.date === getTodayString() && !o.isArchived);
    const revenueToday = todaysOrders.reduce((s, o) => s + (o.amount || 0), 0);
    const contextStr = `Bạn là Trợ lý AI thông minh của phần mềm Quản Lý Doanh Nghiệp (Cty ${company?.name || 'HD'}). 
Tên người dùng đang trò chuyện: ${employee?.name || 'Chưa rõ'}, Chức vụ: ${employee?.position || 'Chưa rõ'}.
Màn hình thao tác hiện tại của người dùng: ${activeTab}. 
Số liệu kinh doanh hôm nay: Tổng doanh thu ${new Intl.NumberFormat('vi-VN').format(revenueToday)} VNĐ, ${todaysOrders.length} đơn hàng mới. 
Tổng số lượng khách hàng trong hệ thống: ${customers.length} người.
Nhiệm vụ của bạn: Trả lời thật ngắn gọn, súc tích, thân thiện. Tư vấn cách làm việc hiệu quả, phân tích nhanh số liệu hoặc hướng dẫn nhân viên cách dùng app.`;

    const apiKey = ""; 
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    
    const payload = {
      contents: newMessages.map(m => ({ role: m.role === 'model' ? 'model' : 'user', parts: [{ text: m.text }] })),
      systemInstruction: { parts: [{ text: contextStr }] }
    };

    const delays = [1000, 2000, 4000, 8000, 16000];
    let attempt = 0;
    let aiResponseText = "Xin lỗi, hệ thống AI đang bận xử lý. Vui lòng thử lại sau giây lát.";

    while (attempt < 6) {
      try {
        const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
        if (!res.ok) throw new Error('API Error');
        const data = await res.json();
        aiResponseText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Mình chưa hiểu rõ ý bạn, hãy thử diễn đạt lại nhé.";
        break;
      } catch (err) {
        if (attempt >= 5) break;
        await new Promise(r => setTimeout(r, delays[attempt]));
        attempt++;
      }
    }

    setMessages([...newMessages, { role: 'model', text: aiResponseText }]);
    setIsLoading(false);
  };

  return (

          <div className="bg-gray-50 w-full h-[85vh] sm:h-[600px] sm:max-w-md sm:rounded-2xl rounded-t-3xl shadow-2xl flex flex-col animate-in slide-in-from-bottom-full duration-300 overflow-hidden border border-gray-200">
            <div className="bg-gradient-to-r from-emerald-600 to-emerald-500 text-white p-4 flex justify-between items-center shrink-0 shadow-sm">

                  <p className="text-[10px] text-emerald-100 font-medium">Trực tuyến • Hỗ trợ {employee?.position}</p>

              </div>
              <button onClick={() => setIsOpen(false)} className="hover:bg-white/20 p-1.5 rounded-full transition-colors"><X size={20} /></button>

                  <Send size={16} className="ml-0.5" />
                </button>

        </div>

// --- APP ENTRY POINT ---
export default function App() {
  const [firebaseUser, setFirebaseUser] = useState(null);
  const [isFirebaseLoading, setIsFirebaseLoading] = useState(true);

  const [currentUser, setCurrentUser] = useState(null); 
  const [currentCompany, setCurrentCompany] = useState(null);
  const [activeTab, setActiveTab] = useState('home');
  const [currentDate, setCurrentDate] = useState(getTodayString());

  // RAW DATA
  const [rawCompanies, setRawCompanies] = useState([]);
  const [rawEmployees, setRawEmployees] = useState([]);
  const [rawAttendance, setRawAttendance] = useState({});
  const [rawFinancials, setRawFinancials] = useState([]);
  const [rawAdvanceRequests, setRawAdvanceRequests] = useState([]); 
  const [rawPerformance, setRawPerformance] = useState({});
  const [rawCustomers, setRawCustomers] = useState([]);
  const [rawProducts, setRawProducts] = useState([]);
  const [rawOrders, setRawOrders] = useState([]);
  const [rawPayments, setRawPayments] = useState([]);
  const [rawExpenses, setRawExpenses] = useState([]); 
  const [rawHolidays, setRawHolidays] = useState([]);

  useEffect(() => {
    if (!isFirebaseConfigured) return;
    const initAuth = async () => {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (err) { console.error("Lỗi xác thực:", err); }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (u) => { setFirebaseUser(u); setIsFirebaseLoading(false); });
    return () => unsubscribe();
  }, []);

  useEffect(() => {
    if (!firebaseUser || !isFirebaseConfigured) return;
    const listeners = [];

    const bindCollection = (colName, setFn, isObject = false, parser = null) => {
      const unsub = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', colName), (snap) => {
        if (isObject) {
          const obj = {};
          snap.forEach(d => { obj[d.id] = parser ? parser(d.data()) : d.data(); });
          setFn(obj);
        } else {
          const arr = [];
          snap.forEach(d => { arr.push(parser ? parser(d.data()) : d.data()); });
          setFn(arr);
        }
      }, console.error);
      listeners.push(unsub);
    };

    bindCollection('companies', setRawCompanies);
    bindCollection('employees', setRawEmployees);
    bindCollection('customers', setRawCustomers);
    bindCollection('products', setRawProducts);
    bindCollection('orders', setRawOrders);
    bindCollection('payments', setRawPayments);
    bindCollection('advances', setRawAdvanceRequests);
    bindCollection('financials', setRawFinancials);
    bindCollection('expenses', setRawExpenses);
    bindCollection('holidays', setRawHolidays); 
    bindCollection('performance', setRawPerformance, true);
    bindCollection('attendance', setRawAttendance, true, (data) => ({
      ...data,
      checkIn: data.checkIn ? new Date(data.checkIn) : null,
      checkOut: data.checkOut ? new Date(data.checkOut) : null,
    }));

    return () => { listeners.forEach(unsub => unsub()); };
  }, [firebaseUser]);

  const myCompanyId = currentUser?.companyId;

  const employees = useMemo(() => rawEmployees.filter(e => e.companyId === myCompanyId), [rawEmployees, myCompanyId]);
  const customers = useMemo(() => rawCustomers.filter(c => c.companyId === myCompanyId), [rawCustomers, myCompanyId]);
  const products = useMemo(() => rawProducts.filter(p => p.companyId === myCompanyId), [rawProducts, myCompanyId]);
  const orders = useMemo(() => rawOrders.filter(o => o.companyId === myCompanyId), [rawOrders, myCompanyId]);
  const payments = useMemo(() => rawPayments.filter(p => p.companyId === myCompanyId), [rawPayments, myCompanyId]);
  const expenses = useMemo(() => rawExpenses.filter(e => e.companyId === myCompanyId), [rawExpenses, myCompanyId]);
  const advanceRequests = useMemo(() => rawAdvanceRequests.filter(a => a.companyId === myCompanyId), [rawAdvanceRequests, myCompanyId]);
  const financials = useMemo(() => rawFinancials.filter(f => f.companyId === myCompanyId), [rawFinancials, myCompanyId]);
  const holidays = useMemo(() => rawHolidays.filter(h => h.companyId === myCompanyId), [rawHolidays, myCompanyId]);
  
  const attendanceRecords = useMemo(() => {
    const filtered = {};
    Object.keys(rawAttendance).forEach(key => {
      if (rawAttendance[key]?.companyId === myCompanyId) filtered[key] = rawAttendance[key];
    });
    return filtered;
  }, [rawAttendance, myCompanyId]);

  const performanceRecords = useMemo(() => {
    const filtered = {};
    Object.keys(rawPerformance).forEach(key => {
      if (rawPerformance[key]?.companyId === myCompanyId) filtered[key] = rawPerformance[key];
    });
    return filtered;
  }, [rawPerformance, myCompanyId]);

  const aggregatedPerformance = useMemo(() => {
    const agg = { ...performanceRecords };
    employees.forEach(emp => {
      if (!agg[emp.id]) agg[emp.id] = { overtime: 0 };
      agg[emp.id].revenue = calculateMonthRevenue(emp.id, orders);
    });
    return agg;
  }, [performanceRecords, orders, employees]);

  const handleRegisterCompany = async (companyName, phone) => {
    if (!firebaseUser) return { success: false, message: "Lỗi kết nối máy chủ." };
    const exist = rawEmployees.find(e => e.phone === phone);
    if (exist) return { success: false, message: "Số điện thoại này đã được đăng ký!" };

    const newCompanyId = `comp_${Date.now()}`;
    const newEmpId = `emp_${Date.now()}`;

    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'companies', newCompanyId), {
      id: newCompanyId, name: companyName, ownerPhone: phone, createdAt: getTodayString(), status: 'trial', expiresAt: new Date(Date.now() + 14 * 24 * 60 * 60 * 1000).toISOString()
    });

    const adminData = {
      id: newEmpId, companyId: newCompanyId, phone: phone, name: 'Chủ doanh nghiệp', position: 'Chủ doanh nghiệp', role: 'super_admin', 
      startDate: getTodayString(), probationDuration: 0, probationUnit: 'days', probationRate: 100,
      basicSalary: 0, supportSalary: 0, experienceSalary: 0, commissionRate: 0, targetRevenue: 0, overtimeRate: 0
    };
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'employees', newEmpId), adminData);

    setCurrentUser({ role: 'super_admin', id: newEmpId, companyId: newCompanyId });
    setCurrentCompany({ id: newCompanyId, name: companyName, status: 'trial' });
    setActiveTab('home');
    return { success: true };
  };

  const handleLogin = (phone) => {
    const emp = rawEmployees.find(e => e.phone === phone);
    if (emp) { 
      const comp = rawCompanies.find(c => c.id === emp.companyId);
      setCurrentUser({ role: emp.role || 'employee', id: emp.id, companyId: emp.companyId }); 
      setCurrentCompany(comp);
      setActiveTab('home'); 
      return true; 
    }
    return false;
  };

  const handleLogout = () => { setCurrentUser(null); setCurrentCompany(null); setActiveTab('home'); };

  const handleCheckIn = async (empId, method) => {
    if (!firebaseUser) return;
    const now = new Date(); const key = `${currentDate}_${empId}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'attendance', key), { companyId: myCompanyId, checkIn: now.toISOString(), checkInMethod: method, status: (now.getHours() >= 8 && now.getMinutes() > 0) ? 'late' : 'present' }, { merge: true });
  };
  
  const handleCheckOut = async (empId, method) => {
    if (!firebaseUser) return;
    const now = new Date(); const key = `${currentDate}_${empId}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'attendance', key), { companyId: myCompanyId, checkOut: now.toISOString(), checkOutMethod: method }, { merge: true });
  };
  
  const handleLeave = async (empId) => {
    if (!firebaseUser) return;
    const key = `${currentDate}_${empId}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'attendance', key), { companyId: myCompanyId, status: 'leave', checkIn: null, checkOut: null }, { merge: true });
  };

  const handleEditAttendance = async (empId, targetDate, editData) => {
    if (!firebaseUser) return;
    const key = `${targetDate}_${empId}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'attendance', key), { 
      ...editData, companyId: myCompanyId,
      checkIn: editData.checkIn ? editData.checkIn.toISOString() : null,
      checkOut: editData.checkOut ? editData.checkOut.toISOString() : null,
      checkInMethod: 'Kế toán điều chỉnh' 
    }, { merge: true });
  };

  const handleAddAdvanceRequest = async (empId, amount, reason) => {
    if (!firebaseUser) return;
    const id = `adv_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'advances', id), { id, companyId: myCompanyId, empId, amount: parseInt(amount), reason, date: getTodayString(), status: 'pending' });
  };
  
  const handleApproveAdvance = async (reqId) => {
    if (!firebaseUser) return;
    const req = advanceRequests.find(r => r.id === reqId);
    if (req && req.status === 'pending') {
      await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'advances', reqId), { status: 'approved' }, { merge: true });
      await addFinancialRecord(req.empId, 'advance', req.amount, req.reason);
    }
  };

  const handleRejectAdvance = async (reqId) => {
    if (!firebaseUser) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'advances', reqId), { status: 'rejected' }, { merge: true });
  };

  const handleToggleArchive = async (collectionName, itemId, currentStatus) => {
    if (!firebaseUser) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', collectionName, itemId), { isArchived: !currentStatus }, { merge: true });
  };

  const handleAddPayment = async (paymentData) => {
    if (!firebaseUser) return;
    const id = `p_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'payments', id), { ...paymentData, id, companyId: myCompanyId, isArchived: false });
  };

  const handleDeletePayment = async (paymentId) => {
    if (!firebaseUser) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'payments', paymentId));
  };

  const handleAddExpense = async (empId, expenseData) => {
    if (!firebaseUser) return;
    const id = `exp_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'expenses', id), { ...expenseData, empId, id, companyId: myCompanyId, isArchived: false });
  };

  const handleDeleteExpense = async (expenseId) => {
    if (!firebaseUser) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'expenses', expenseId));
  };

  const addFinancialRecord = async (empId, type, amount, reason) => {
    if (!firebaseUser) return;
    const id = `f_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'financials', id), { id, companyId: myCompanyId, empId, type, amount: parseInt(amount), reason, date: getTodayString(), isArchived: false });
  };

  const updatePerformance = async (empId, overtime) => {
    if (!firebaseUser) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'performance', empId), { companyId: myCompanyId, overtime: parseFloat(overtime) || 0 }, { merge: true });
  };

  const handleAddEmployee = async (newEmpData) => {
    if (!firebaseUser) return;
    const id = Date.now().toString();
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'employees', id), { ...newEmpData, id, companyId: myCompanyId });
  };

  const handleEditEmployee = async (empId, updatedData) => {
    if (!firebaseUser) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'employees', empId), updatedData, { merge: true });
  };

  const handleAddCustomer = async (empId, customerData) => {
    if (!firebaseUser) return;
    const id = `c_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'customers', id), { ...customerData, empId, id, companyId: myCompanyId, isArchived: false });
  };

  const handleAddProduct = async (productData) => {
    if (!firebaseUser) return;
    const id = `prod_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'products', id), { ...productData, id, companyId: myCompanyId, isArchived: false });
  };

  const handleEditProduct = async (prodId, updatedData) => {
    if (!firebaseUser) return;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'products', prodId), updatedData, { merge: true });
  };

  const handleDeleteProduct = async (prodId) => {
    if (!firebaseUser) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'products', prodId));
  };

  const handleAddOrder = async (empId, orderData) => {
    if (!firebaseUser) return;
    const id = `o_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'orders', id), { ...orderData, empId, id, companyId: myCompanyId, isArchived: false });
  };

  const handleDeleteOrder = async (orderId) => {
    if (!firebaseUser) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'orders', orderId));
  };

  const handleAddHoliday = async (holidayData) => {
    if (!firebaseUser) return;
    const id = `hol_${Date.now()}`;
    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'holidays', id), { ...holidayData, id, companyId: myCompanyId });
  };

  const handleDeleteHoliday = async (holidayId) => {
    if (!firebaseUser) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'holidays', holidayId));
  };

  if (!isFirebaseConfigured) return <MissingFirebaseError />;
  if (isFirebaseLoading) return <div className="flex h-screen items-center justify-center bg-gray-50"><p className="text-emerald-600 font-medium flex items-center animate-pulse"><CalendarDays className="mr-2 animate-spin"/> Đang kết nối Đám mây...</p></div>;
  
  if (!currentUser) {
    return (
      <>
        <style dangerouslySetInnerHTML={{__html: `@import url('https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@400;500;600;700;800&display=swap'); body { font-family: 'Plus Jakarta Sans', sans-serif !important; }`}} />
        <LoginRegisterView onLogin={handleLogin} onRegister={handleRegisterCompany} />
      </>
    );
  }

  const employeeInfo = employees.find(e => e.id === currentUser.id);

  return (
    <>
      <style dangerouslySetInnerHTML={{__html: `@import url('https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@400;500;600;700;800&display=swap'); body { font-family: 'Plus Jakarta Sans', sans-serif !important; }`}} />
      <MainAppView 
        currentUser={currentUser} employee={employeeInfo} currentCompany={currentCompany} activeTab={activeTab} setActiveTab={setActiveTab}
        employees={employees} attendance={attendanceRecords} date={currentDate} financials={financials} performance={aggregatedPerformance}
        customers={customers} orders={orders} payments={payments} products={products} advanceRequests={advanceRequests} expenses={expenses} holidays={holidays}
        onCheckIn={handleCheckIn} onCheckOut={handleCheckOut} onLeave={handleLeave} onLogout={handleLogout}
        onAddCustomer={handleAddCustomer} onAddOrder={handleAddOrder} onDeleteOrder={handleDeleteOrder}
        onAddPayment={handleAddPayment} onDeletePayment={handleDeletePayment}
        onAddExpense={handleAddExpense} onDeleteExpense={handleDeleteExpense}
        onAddAdvanceRequest={handleAddAdvanceRequest} onEditAttendance={handleEditAttendance}
        onAddFinancial={addFinancialRecord} onUpdatePerformance={updatePerformance}
        onApproveAdvance={handleApproveAdvance} onRejectAdvance={handleRejectAdvance}
        onAddEmployee={handleAddEmployee} onEditEmployee={handleEditEmployee}
        onOverrideCheckIn={handleCheckIn} onOverrideCheckOut={handleCheckOut}
        onAddProduct={handleAddProduct} onEditProduct={handleEditProduct} onDeleteProduct={handleDeleteProduct}
        onAddHoliday={handleAddHoliday} onDeleteHoliday={handleDeleteHoliday}
        onToggleArchiveOrder={(id, status) => handleToggleArchive('orders', id, status)}
        onToggleArchivePayment={(id, status) => handleToggleArchive('payments', id, status)}
        onToggleArchiveExpense={(id, status) => handleToggleArchive('expenses', id, status)}
        onToggleArchiveProduct={(id, status) => handleToggleArchive('products', id, status)}
      />
    </>
  );
}

// --- MAIN LAYOUT ---
function MainAppView({ 
  currentUser, employee, currentCompany, activeTab, setActiveTab, employees, attendance, date, financials, performance, customers, orders, payments, products, advanceRequests, expenses, holidays,
  onCheckIn, onCheckOut, onLeave, onLogout, onAddCustomer, onAddOrder, onDeleteOrder, onAddPayment, onDeletePayment, onAddExpense, onDeleteExpense, onAddAdvanceRequest, onEditAttendance, onAddFinancial, onUpdatePerformance, onApproveAdvance, onRejectAdvance, onAddEmployee, onEditEmployee, onOverrideCheckIn, onOverrideCheckOut, onAddProduct, onEditProduct, onDeleteProduct, onAddHoliday, onDeleteHoliday,
  onToggleArchiveOrder, onToggleArchivePayment, onToggleArchiveExpense, onToggleArchiveProduct
}) {
  const isSuperAdmin = currentUser?.role === 'super_admin';
  const position = employee?.position || '';
  const isAccounting = isSuperAdmin || ['Kế toán nhân sự', 'Kế toán', 'Admin'].includes(position);
  const isSales = ['Kinh doanh', 'Bán hàng'].includes(position);
  const isDriver = position === 'Tài xế';

  const [showAuthModal, setShowAuthModal] = useState(false);
  const [authType, setAuthType] = useState('in'); 
  const [authMethod, setAuthMethod] = useState('GPS');
  
  const [showAdvanceModal, setShowAdvanceModal] = useState(false);
  const [advanceAmount, setAdvanceAmount] = useState('');
  const [advanceReason, setAdvanceReason] = useState('');

  const record = attendance[`${date}_${employee?.id}`];
  const isCheckedIn = record?.checkIn; 
  const isCheckedOut = record?.checkOut; 
  const isLeave = record?.status === 'leave';

  const details = calculateSalaryDetails(employee?.id, employees, attendance, financials, performance, customers, orders, payments, holidays) || { netSalary: 0, workDays: 0, workDaysProbation: 0, workDaysOfficial: 0, expMonths: 0, baseSalaryCalc: 0, supportSalary: 0, experienceSalary: 0, commission: 0, tieredBonus: 0, overtimePay: 0, totalBonus: 0, holidayBonus: 0, totalPenalty: 0, totalAdvance: 0, badDebt: 0, perf: {overtime: 0} };

  const handleConfirmAuth = () => {
    setShowAuthModal(false); 
    if (authType === 'in') onCheckIn(employee.id, authMethod); else onCheckOut(employee.id, authMethod);
  };
  
  const handleAdvanceSubmit = (e) => {
    e.preventDefault();
    if(advanceAmount && parseInt(parseInputCurrency(advanceAmount)) > 0) {
      onAddAdvanceRequest(employee.id, parseInputCurrency(advanceAmount), advanceReason);
      setShowAdvanceModal(false);
      setAdvanceAmount(''); setAdvanceReason('');
    }
  };

  const renderHeader = () => {
    if (activeTab === 'home') return null;
    if (activeTab === 'more') {
      return (
        <header className="bg-emerald-500 text-white p-4 shadow-sm shrink-0">
          <h1 className="text-xl font-bold">Thêm</h1>
        </header>
      );
    }
    return (
      <header className="bg-gradient-to-r from-emerald-500 to-emerald-600 text-white p-4 shadow-sm shrink-0">
        <div className="flex justify-between items-center">
          <div className="flex items-center gap-3">
            <button onClick={() => setActiveTab('home')} className="hover:bg-emerald-700/50 p-1.5 rounded-full transition"><ChevronLeft size={24} /></button>
            <h1 className="text-lg font-bold">
              {activeTab === 'profile' ? 'Cá nhân' : 
               activeTab === 'customers' ? 'Khách hàng' : 
               activeTab === 'orders' ? 'Đơn hàng' : 
               activeTab === 'products' ? 'Sản phẩm' : 
               activeTab === 'debt' ? 'Sổ nợ' : 
               activeTab === 'finance' ? 'Thu chi' : 
               activeTab === 'company_attendance' ? 'Chấm công' : 
               activeTab === 'payroll' ? 'Bảng lương' : 
               activeTab === 'employees' ? 'Nhân sự' : 
               activeTab === 'settings' ? 'Cấu hình Lễ Tết' : 
               activeTab === 'billing' ? 'Gói cước' : 'Quản lý'}
            </h1>
          </div>
          <div className="flex items-center gap-3">
            <Search size={20} />
            <Filter size={20} />
          </div>
        </div>
      </header>
    );
  };

  const renderContent = () => {
    switch (activeTab) {
      case 'home': return <DashboardView employee={employee} company={currentCompany} employees={employees} attendance={attendance} date={date} orders={orders} setActiveTab={setActiveTab} />;
      case 'profile': return (
        <div className="space-y-4 animate-in fade-in zoom-in-95">
          <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
            <div className="flex justify-between items-center mb-4">
              <h2 className="text-base font-bold text-gray-800">Chấm công hôm nay</h2>
              <span className="text-sm font-medium text-gray-500">{new Date(date).toLocaleDateString('vi-VN')}</span>
            </div>
            
            <div className="grid grid-cols-2 gap-3 text-center mb-5">
              <div className="bg-gray-50 p-3 rounded-xl border border-gray-100"><p className="text-xs text-gray-500 mb-1">Giờ vào</p><p className="font-bold text-xl text-gray-800">{formatTime(record?.checkIn)}</p></div>
              <div className="bg-gray-50 p-3 rounded-xl border border-gray-100"><p className="text-xs text-gray-500 mb-1">Giờ ra</p><p className="font-bold text-xl text-gray-800">{formatTime(record?.checkOut)}</p></div>
            </div>

            {!isCheckedIn && !isLeave && !isSuperAdmin ? (
              <div className="flex gap-3">
                <button onClick={() => {setAuthType('in'); setShowAuthModal(true);}} className="flex-1 bg-emerald-500 text-white py-3 rounded-xl font-semibold shadow-sm hover:bg-emerald-600">Vào ca</button>
                <button onClick={() => onLeave(employee.id)} className="flex-1 bg-white border border-gray-300 text-gray-700 py-3 rounded-xl font-semibold">Xin nghỉ</button>
              </div>
            ) : isCheckedIn && !isCheckedOut && !isLeave && !isSuperAdmin ? (
              <button onClick={() => {setAuthType('out'); setShowAuthModal(true);}} className="w-full bg-orange-500 text-white py-3 rounded-xl font-semibold shadow-sm">Ra ca</button>
            ) : <div className="bg-gray-50 text-emerald-600 p-3 rounded-xl text-center text-sm font-semibold border border-emerald-100">{isSuperAdmin ? 'Admin không cần chấm công' : isLeave ? 'Đang nghỉ phép' : 'Đã hoàn thành ca'}</div>}
            
          </div>

          {!isSuperAdmin && (
            <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
              <div className="flex justify-between items-center mb-4">
                <h2 className="font-bold text-gray-800 text-base">Bảng Lương Dự Kiến</h2>
                <button onClick={() => setShowAdvanceModal(true)} className="text-orange-600 bg-orange-50 px-3 py-1.5 rounded-lg text-xs font-bold flex items-center"><Banknote size={14} className="mr-1"/> Ứng lương</button>
              </div>
              <div className="text-center mb-5 pb-5 border-b border-dashed border-gray-200">
                <p className="text-xs text-gray-500 mb-1 uppercase tracking-wide">Thực lãnh</p>
                <p className={`text-4xl font-bold ${details.netSalary < 0 ? 'text-red-500' : 'text-emerald-600'}`}>{formatCurrency(details.netSalary)} đ</p>
              </div>
              <div className="space-y-3">
                <div>
                  <h4 className="text-[10px] font-bold text-emerald-600 uppercase mb-2">Thu nhập (+)</h4>
                  <div className="space-y-1.5 px-1 text-sm">
                    <div className="flex justify-between flex-wrap"><span className="text-gray-500">Lương CB ({details.workDays} ngày)</span> <span className="font-medium text-right">{formatCurrency(details.baseSalaryCalc)}</span>
                      {details.workDaysProbation > 0 && <span className="w-full text-right text-[10px] text-gray-400 mt-0.5">({details.workDaysProbation} ngày thử việc, {details.workDaysOfficial} ngày chính thức)</span>}
                    </div>
                    <div className="flex justify-between"><span className="text-gray-500">Kinh nghiệm ({details.expMonths} tháng)</span> <span className="font-medium text-blue-600">+{formatCurrency(details.experienceSalary)}</span></div>
                    {isSales && (<div className="flex justify-between"><span className="text-gray-500">Hoa hồng</span> <span className="font-medium text-emerald-600">+{formatCurrency(details.commission)}</span></div>)}
                    <div className="flex justify-between"><span className="text-gray-500">Thưởng / Tăng ca</span> <span className="font-medium text-emerald-600">+{formatCurrency(details.overtimePay + details.totalBonus - details.holidayBonus)}</span></div>
                    {details.holidayBonus > 0 && <div className="flex justify-between"><span className="text-gray-500">Thưởng Lễ Tết tự động</span> <span className="font-medium text-purple-600">+{formatCurrency(details.holidayBonus)}</span></div>}
                  </div>
                </div>
                <div>
                  <h4 className="text-[10px] font-bold text-red-500 uppercase mb-2">Khấu trừ (-)</h4>
                  <div className="space-y-1.5 px-1 text-sm">
                    <div className="flex justify-between"><span className="text-gray-500">Tạm ứng / Phạt</span> <span className="font-medium text-red-500">-{formatCurrency(details.totalAdvance + details.totalPenalty)}</span></div>
                    {isSales && details.badDebt > 0 && (<div className="flex justify-between"><span className="text-red-500 font-medium">Khấu trừ nợ</span> <span className="font-medium text-red-500">-{formatCurrency(details.badDebt)}</span></div>)}
                  </div>
                </div>
              </div>
            </div>
          )}
        </div>
      );
      case 'settings': return <SettingsView isAccounting={isAccounting} holidays={holidays} onAddHoliday={onAddHoliday} onDeleteHoliday={onDeleteHoliday} />;
      case 'billing': return <BillingView company={currentCompany} />;
      case 'company_attendance': return <AttendanceView employees={employees} attendance={attendance} date={date} onEditAttendance={onEditAttendance} onOverrideCheckIn={(id)=>onOverrideCheckIn(id, 'Kế toán chấm hộ')} onOverrideCheckOut={(id)=>onOverrideCheckOut(id, 'Kế toán chấm hộ')} />;
      case 'products': return <ProductManagementView isAccounting={isAccounting} products={products} onAddProduct={onAddProduct} onEditProduct={onEditProduct} onToggleArchiveProduct={onToggleArchiveProduct} onDeleteProduct={onDeleteProduct} />;
      case 'employees': return <EmployeeView employees={employees} onAddEmployee={onAddEmployee} onEditEmployee={onEditEmployee} isSuperAdmin={isSuperAdmin} />;
      case 'customers': return <CustomerCRMView employee={employee} customers={isSales ? customers.filter(c=>c.empId===employee.id) : customers} orders={orders} payments={payments} onAddCustomer={onAddCustomer} employees={employees} />;
      case 'orders': return <OrderManagementView isAccounting={isAccounting} employee={employee} employees={employees} customers={customers} orders={orders} products={products} onAddOrder={onAddOrder} onToggleArchiveOrder={onToggleArchiveOrder} onDeleteOrder={onDeleteOrder} onAddPayment={onAddPayment} onAddCustomer={onAddCustomer} onAddExpense={(data) => onAddExpense(employee?.id || 'admin', data)} />;
      case 'debt': return <DebtManagementView isAccounting={isAccounting} isDriver={isDriver} employee={employee} customers={customers} orders={orders} payments={payments} employees={employees} onAddPayment={onAddPayment} onDeletePayment={onDeletePayment} />;
      case 'finance': return <FinanceView isAccounting={isAccounting} employee={employee} expenses={expenses} payments={payments} onAddExpense={(data) => onAddExpense(employee.id, data)} onAddPayment={onAddPayment} onToggleArchiveExpense={onToggleArchiveExpense} onToggleArchivePayment={onToggleArchivePayment} onDeleteExpense={onDeleteExpense} onDeletePayment={onDeletePayment} employees={employees} customers={customers} />;
      case 'payroll': return <SalaryView employees={employees} attendance={attendance} financials={financials} performance={performance} customers={customers} orders={orders} payments={payments} holidays={holidays} onAddFinancial={onAddFinancial} onUpdatePerformance={onUpdatePerformance} advanceRequests={advanceRequests} onApproveAdvance={onApproveAdvance} onRejectAdvance={onRejectAdvance} />;
      case 'more': return <MoreMenu isAccounting={isAccounting} isSales={isSales} isDriver={isDriver} isSuperAdmin={isSuperAdmin} setActiveTab={setActiveTab} onLogout={onLogout} employee={employee} />;
      default: return <DashboardView employee={employee} company={currentCompany} employees={employees} attendance={attendance} date={date} orders={orders} setActiveTab={setActiveTab} />;
    }
  };

  return (
    <div className="flex flex-col h-screen bg-[#f4f6f8] w-full max-w-md mx-auto shadow-2xl relative overflow-hidden font-sans">
      {renderHeader()}
      
      <main className="flex-1 overflow-y-auto pb-24 px-4 pt-4">{renderContent()}</main>

      {showAuthModal && (
        <div className="fixed inset-0 bg-black/60 flex items-end sm:items-center justify-center z-50 p-0 sm:p-4">
          <div className="bg-white rounded-t-3xl sm:rounded-3xl p-6 w-full max-w-md animate-in slide-in-from-bottom-10">
            <div className="flex justify-between items-center mb-6"><h3 className="font-bold text-xl text-gray-800">{authType === 'in' ? 'Vào ca' : 'Ra ca'}</h3><button onClick={() => setShowAuthModal(false)}><X/></button></div>
            <button onClick={handleConfirmAuth} className="w-full bg-emerald-500 hover:bg-emerald-600 text-white py-3.5 rounded-xl font-bold shadow-sm transition-colors">Xác nhận</button>
          </div>
        </div>
      )}

      {showAdvanceModal && (
        <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl p-5 w-full max-w-sm animate-in zoom-in-95">
            <div className="flex justify-between items-center mb-4"><h3 className="font-bold text-lg text-gray-800">Yêu Cầu Tạm Ứng</h3><button onClick={() => setShowAdvanceModal(false)} className="text-gray-400"><X /></button></div>
            <form onSubmit={handleAdvanceSubmit} className="space-y-4">
              <div><label className="block text-xs font-semibold text-gray-700 mb-1">Số tiền (VNĐ)</label><input type="tel" required value={formatInputCurrency(advanceAmount)} onChange={(e) => setAdvanceAmount(parseInputCurrency(e.target.value))} className="w-full border border-gray-300 rounded-lg p-2.5 outline-none focus:ring-2 focus:ring-emerald-500" placeholder="VD: 2.000.000" /></div>
              <div><label className="block text-xs font-semibold text-gray-700 mb-1">Lý do</label><input type="text" required value={advanceReason} onChange={(e) => setAdvanceReason(capitalizeFirst(e.target.value))} className="w-full border border-gray-300 rounded-lg p-2.5 outline-none focus:ring-2 focus:ring-emerald-500" placeholder="Lý do cá nhân..." /></div>
              <button type="submit" className="w-full bg-orange-500 hover:bg-orange-600 text-white py-3 rounded-lg font-semibold mt-4 transition-colors">Gửi Yêu Cầu</button>
            </form>
          </div>
        </div>
      )}

      <SmartAIAssistant employee={employee} company={currentCompany} activeTab={activeTab} customers={customers} orders={orders} expenses={expenses} />

      <nav className="absolute bottom-0 left-0 right-0 bg-white border-t border-gray-100 z-40 pb-safe shadow-[0_-4px_10px_rgba(0,0,0,0.03)]">
        <div className="flex justify-around items-center px-1 py-1.5">
          {isAccounting && <NavButton icon={<Store />} label="Quản lý" active={activeTab === 'home'} onClick={() => setActiveTab('home')} />}
          {(isAccounting || isSales) && <NavButton icon={<ClipboardList />} label="Đơn hàng" active={activeTab === 'orders'} onClick={() => setActiveTab('orders')} />}
          {(isAccounting || isDriver) && <NavButton icon={<BookText />} label="Sổ nợ" active={activeTab === 'debt'} onClick={() => setActiveTab('debt')} />}

function FinanceView({ isAccounting, employee, expenses, payments, onAddExpense, onAddPayment, onToggleArchiveExpense, onToggleArchivePayment, onDeleteExpense, onDeletePayment, employees, customers }) {
  const [filterMonth, setFilterMonth] = useState(getTodayString().substring(0, 7));
  const [showExpenseModal, setShowExpenseModal] = useState(false);
  const [showPaymentModal, setShowPaymentModal] = useState(false);
  const [actionItem, setActionItem] = useState(null);

  const [newExpense, setNewExpense] = useState({ category: 'Chi mua hàng hóa', amount: '', note: '', date: getTodayString() });
  const [newPayment, setNewPayment] = useState({ customerId: '', amount: '', note: 'Thu nợ', date: getTodayString() });

  const categories = ['Chi mua hàng hóa', 'Chi xăng dầu', 'Chi phí văn phòng', 'Chi phí khác'];

  const allTransactions = useMemo(() => {
    const list = [
      ...expenses.map(e => ({ ...e, isExpense: true })),
      ...payments.map(p => ({ ...p, isExpense: false }))
    ];
    return list.filter(t => t.date.startsWith(filterMonth) && !t.isArchived).sort((a,b) => new Date(b.date) - new Date(a.date));
  }, [expenses, payments, filterMonth]);

  const totalExpense = allTransactions.filter(t => t.isExpense).reduce((sum, t) => sum + (t.amount || 0), 0);
  const totalIncome = allTransactions.filter(t => !t.isExpense).reduce((sum, t) => sum + (t.amount || 0), 0);
  const balance = totalIncome - totalExpense;

  const handleExpenseSubmit = (e) => {
    e.preventDefault();
    if (newExpense.amount > 0) onAddExpense({ ...newExpense, note: capitalizeFirst(newExpense.note) });
    setShowExpenseModal(false);
    setNewExpense({ category: 'Chi mua hàng hóa', amount: '', note: '', date: getTodayString() });
  };
  const handlePaymentSubmit = (e) => {
    e.preventDefault();
    if (newPayment.amount > 0 && newPayment.customerId) onAddPayment({ ...newPayment, note: capitalizeFirst(newPayment.note) });
    setShowPaymentModal(false);
    setNewPayment({ customerId: '', amount: '', note: 'Thu nợ', date: getTodayString() });
  };

  return (
    <div className="space-y-4 animate-in fade-in pb-16">
      <div className="flex gap-2 mb-4">
        <select className="bg-white border border-gray-200 rounded-full px-3 py-1.5 text-xs font-semibold outline-none" value={filterMonth} onChange={e=>setFilterMonth(e.target.value)}>
          <option value={getTodayString().substring(0, 7)}>Tháng này</option>
          <option value="">Tất cả</option>
        </select>
        <button className="bg-white border border-gray-200 rounded-full px-3 py-1.5 text-xs font-semibold outline-none flex items-center gap-1">Khách hàng <ChevronDown size={12}/></button>
      </div>

      <div className="bg-white rounded-xl shadow-sm border border-gray-50 p-4">
        <div className="flex justify-between items-end mb-4 border-b border-gray-100 pb-4">
          <span className="text-xs font-bold text-gray-500 uppercase">Chênh lệch</span>
          <span className="text-2xl font-bold text-gray-800">{formatCurrency(balance)} đ</span>
        </div>
        <div className="grid grid-cols-2 gap-3">
          <div className="bg-orange-50/50 border border-orange-100 rounded-lg p-3">
            <p className="text-[10px] text-orange-800 font-bold mb-1 flex items-center gap-1"><ArrowRightLeft size={12} className="rotate-90"/> TỔNG CHI</p>
            <p className="font-bold text-orange-600">{formatCurrency(totalExpense)}</p>
          </div>
          <div className="bg-emerald-50/50 border border-emerald-100 rounded-lg p-3">
            <p className="text-[10px] text-emerald-800 font-bold mb-1 flex items-center gap-1"><ArrowRightLeft size={12} className="-rotate-90"/> TỔNG THU</p>
            <p className="font-bold text-emerald-600">{formatCurrency(totalIncome)}</p>
          </div>
        </div>
      </div>

      <div className="space-y-0 bg-white rounded-xl border border-gray-50 shadow-sm overflow-hidden">
        {allTransactions.length === 0 ? <p className="p-4 text-center text-sm text-gray-400">Chưa có giao dịch.</p> : 
          allTransactions.map(t => {
            const isChi = t.isExpense;
            const empName = employees.find(e => e.id === t.empId)?.name || 'Admin';
            const cusName = isChi ? null : customers.find(c => c.id === t.customerId)?.name || 'Khách vô danh';
            return (
              <div key={t.id} className="p-3 border-b border-gray-50 last:border-0 relative group flex items-start gap-3 bg-white">
                {isAccounting && <button onClick={() => { if(window.confirm('Xóa vĩnh viễn giao dịch này?')) isChi ? onDeleteExpense(t.id) : onDeletePayment(t.id); }} className="absolute top-3 right-3 text-gray-300 hover:text-red-500"><Trash2 size={14}/></button>}
                <div className={`mt-1 w-6 h-6 rounded-full flex items-center justify-center shrink-0 border ${isChi ? 'border-orange-200 text-orange-500 bg-orange-50' : 'border-emerald-200 text-emerald-500 bg-emerald-50'}`}>
                  {isChi ? <MinusCircle size={14}/> : <PlusCircle size={14}/>}
                </div>
                <div className="flex-1 pr-6">
                  <div className="flex justify-between items-start mb-1">
                    <h3 className="font-bold text-sm text-gray-800 line-clamp-1">{isChi ? t.note : cusName}</h3>
                    <span className={`font-bold shrink-0 ${isChi ? 'text-orange-600' : 'text-emerald-600'}`}>{isChi ? '-' : '+'}{formatCurrency(t.amount)}</span>
                  </div>
                  <p className="text-xs text-gray-600 mb-0.5">{isChi ? t.category : t.note}</p>
                  <p className="text-[10px] text-gray-400">{new Date(t.date).toLocaleDateString('vi-VN')} • Tạo bởi: {empName}</p>
                </div>
              </div>
            )
          })
        }
      </div>

      <div className="fixed bottom-[70px] left-0 w-full px-4 flex gap-3 pointer-events-none">
        <div className="w-full max-w-md mx-auto flex gap-3 pointer-events-auto">
          <button onClick={() => setShowExpenseModal(true)} className="flex-1 bg-orange-500 text-white rounded-xl py-3 font-bold shadow-lg flex items-center justify-center gap-2 hover:bg-orange-600"><MinusCircle size={18}/> Khoản chi</button>
          <button onClick={() => setShowPaymentModal(true)} className="flex-1 bg-emerald-500 text-white rounded-xl py-3 font-bold shadow-lg flex items-center justify-center gap-2 hover:bg-emerald-600"><PlusCircle size={18}/> Khoản thu</button>
        </div>
      </div>

      {showExpenseModal && (
        <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl p-5 w-full max-w-sm animate-in zoom-in-95">
            <h3 className="font-bold text-lg mb-4">Tạo Khoản Chi</h3>
            <form onSubmit={handleExpenseSubmit} className="space-y-3">
              <select value={newExpense.category} onChange={e=>setNewExpense({...newExpense, category: e.target.value})} className="w-full border p-2.5 rounded-xl text-sm outline-none bg-white">{categories.map(c=><option key={c}>{c}</option>)}</select>
              <input required type="tel" value={formatInputCurrency(newExpense.amount)} onChange={e=>setNewExpense({...newExpense, amount: parseInputCurrency(e.target.value)})} placeholder="Số tiền" className="w-full border p-2.5 rounded-xl text-sm" />
              <input required type="text" value={newExpense.note} onChange={e=>setNewExpense({...newExpense, note: capitalizeFirst(e.target.value)})} placeholder="Nội dung" className="w-full border p-2.5 rounded-xl text-sm" />
              <div className="flex gap-2">
                <button type="button" onClick={()=>setShowExpenseModal(false)} className="flex-1 bg-gray-100 py-3 rounded-xl text-gray-700 font-bold">Hủy</button>
                <button type="submit" className="flex-1 bg-orange-500 text-white py-3 rounded-xl font-bold">Lưu chi</button>
              </div>
            </form>
          </div>
        </div>
      )}

      {showPaymentModal && (
        <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl p-5 w-full max-w-sm animate-in zoom-in-95">
            <h3 className="font-bold text-lg mb-4">Tạo Khoản Thu</h3>
            <form onSubmit={handlePaymentSubmit} className="space-y-3">
              <select required value={newPayment.customerId} onChange={e=>setNewPayment({...newPayment, customerId: e.target.value})} className="w-full border p-2.5 rounded-xl text-sm outline-none bg-white">
                <option value="">-- Chọn khách hàng --</option>
                {customers.map(c=><option key={c.id} value={c.id}>{c.name}</option>)}
              </select>
              <input required type="tel" value={formatInputCurrency(newPayment.amount)} onChange={e=>setNewPayment({...newPayment, amount: parseInputCurrency(e.target.value)})} placeholder="Số tiền" className="w-full border p-2.5 rounded-xl text-sm" />
              <input required type="text" value={newPayment.note} onChange={e=>setNewPayment({...newPayment, note: capitalizeFirst(e.target.value)})} placeholder="Nội dung (Mặc định: Thu nợ)" className="w-full border p-2.5 rounded-xl text-sm" />
              <div className="flex gap-2">
                <button type="button" onClick={()=>setShowPaymentModal(false)} className="flex-1 bg-gray-100 py-3 rounded-xl text-gray-700 font-bold">Hủy</button>
                <button type="submit" className="flex-1 bg-emerald-500 text-white py-3 rounded-xl font-bold">Lưu thu</button>
              </div>
            </form>
          </div>
        </div>
      )}
    </div>
  );
}

    </button>
  ); 
}

function OrderManagementView({ isAccounting, employee, employees, customers, orders, products, onAddOrder, onToggleArchiveOrder, onDeleteOrder, onAddPayment, onAddCustomer, onAddExpense }) {
  const [showAddOrder, setShowAddOrder] = useState(false); 
  const [showQuickAddCus, setShowQuickAddCus] = useState(false);
  const [tab, setTab] = useState('all');
  const [searchCus, setSearchCus] = useState('');
  const [showCusDropdown, setShowCusDropdown] = useState(false);
  const [errorMsg, setErrorMsg] = useState('');

  const [newOrder, setNewOrder] = useState({ 
    customerId: '', 
    items: [{ description: '', quantity: 1, unitPrice: '' }], 
    discount: '', 
    extraExpenseName: '',
    extraExpenseAmount: '',
    extraExpensePayer: 'buyer', 
    date: getTodayString(), 
    upfrontPayment: '', 
    paymentMethod: 'Chuyển khoản' 
  });
  
  const [newCus, setNewCus] = useState({ name: '', phone: '', address: '' });

  const activeOrders = orders.filter(o => !o.isArchived);
  const filteredCustomers = customers.filter(c => (c.name || '').toLowerCase().includes(searchCus.toLowerCase()) || (c.phone || '').includes(searchCus));
  
  const subTotal = newOrder.items.reduce((sum, item) => sum + (parseFloat(item.quantity)||0) * (parseFloat(item.unitPrice)||0), 0);
  const discountAmount = parseFloat(newOrder.discount) || 0;
  const extraExpense = parseFloat(newOrder.extraExpenseAmount) || 0;
  
  let customerExtraExpense = 0;
  let sellerExtraExpense = 0;
  
  if (newOrder.extraExpensePayer === 'buyer') {
    customerExtraExpense = extraExpense;
  } else if (newOrder.extraExpensePayer === 'shared') {
    customerExtraExpense = extraExpense / 2;
    sellerExtraExpense = extraExpense / 2;
  } else if (newOrder.extraExpensePayer === 'seller') {
    sellerExtraExpense = extraExpense;
  }

  const totalAmount = Math.max(0, subTotal - discountAmount + customerExtraExpense);

  const handleAddItem = () => setNewOrder({...newOrder, items: [...newOrder.items, { description: '', quantity: 1, unitPrice: '' }]});
  const handleRemoveItem = (index) => {
    const newItems = [...newOrder.items]; newItems.splice(index, 1);
    setNewOrder({...newOrder, items: newItems});
  };
  const handleItemChange = (index, field, value) => {
    const newItems = [...newOrder.items]; newItems[index][field] = value;
    setNewOrder({...newOrder, items: newItems});
  };
  const handleItemProductChange = (index, prodId) => {
    const prod = products?.find(p => p.id === prodId);
    const newItems = [...newOrder.items]; 
    if(prod) { newItems[index].description = prod.name; newItems[index].unitPrice = prod.sellingPrice; } 
    else { newItems[index].description = ''; newItems[index].unitPrice = ''; }
    setNewOrder({...newOrder, items: newItems});
  };

  const handleQuickAddCustomer = (e) => {
    e.preventDefault();
    const finalEmpId = ['Kinh doanh', 'Bán hàng'].includes(employee?.position) ? employee.id : employee?.id;
    if (!finalEmpId) return setErrorMsg("Vui lòng chọn nhân viên phụ trách khách hàng này!");
    
    onAddCustomer(finalEmpId, { name: newCus.name, phone: newCus.phone, address: newCus.address });
    setSearchCus(newCus.name);
    setShowQuickAddCus(false);
    setNewCus({name:'', phone:'', address:''});
  };

  const handleAddSubmit = (e) => {
    e.preventDefault(); 
    setErrorMsg('');
    if (!newOrder.customerId) { setErrorMsg("Vui lòng chọn khách hàng!"); return; }
    const validItems = newOrder.items.filter(i => i.description.trim() !== '');
    if (validItems.length === 0) { setErrorMsg("Vui lòng nhập ít nhất 1 sản phẩm!"); return; }

    const empId = employee?.id || 'admin';

    onAddOrder(empId, { 
      customerId: newOrder.customerId, 
      items: validItems, 
      discount: discountAmount, 
      extraExpenseName: capitalizeFirst(newOrder.extraExpenseName),
      extraExpenseAmount: extraExpense,
      extraExpensePayer: newOrder.extraExpensePayer,
      customerExtraExpense: customerExtraExpense,
      sellerExtraExpense: sellerExtraExpense,
      amount: totalAmount, 
      date: newOrder.date 
    });

    const upfront = parseFloat(newOrder.upfrontPayment) || 0;
    if (upfront > 0 && onAddPayment) {
      onAddPayment({ customerId: newOrder.customerId, amount: upfront, note: `Khách thanh toán trước - ${newOrder.paymentMethod}`, date: newOrder.date });
    }

    if (sellerExtraExpense > 0 && onAddExpense) {
      const cusName = customers.find(c => c.id === newOrder.customerId)?.name || 'Khách hàng';
      onAddExpense({
        category: 'Chi phí khác',
        amount: sellerExtraExpense,
        note: `Phụ phí đơn hàng (${capitalizeFirst(newOrder.extraExpenseName) || 'Không tên'}) - Khách: ${cusName}`,
        date: newOrder.date
      });
    }

    setShowAddOrder(false); 
    setSearchCus('');
    setNewOrder({ customerId: '', items: [{ description: '', quantity: 1, unitPrice: '' }], discount: '', extraExpenseName: '', extraExpenseAmount: '', extraExpensePayer: 'buyer', date: getTodayString(), upfrontPayment: '', paymentMethod: 'Chuyển khoản' });
  };

  return (
    <div className="space-y-4 animate-in fade-in pb-16">
      <div className="flex border-b border-gray-200">
        <button onClick={()=>setTab('all')} className={`flex-1 py-2 text-sm font-semibold border-b-2 ${tab==='all'?'border-emerald-500 text-emerald-600':'border-transparent text-gray-500'}`}>Tất cả</button>
        <button onClick={()=>setTab('pending')} className={`flex-1 py-2 text-sm font-semibold border-b-2 ${tab==='pending'?'border-emerald-500 text-emerald-600':'border-transparent text-gray-500'}`}>Chờ xác nhận</button>
      </div>
      
      <div className="space-y-3">
        {activeOrders.reverse().map(o => {
          const cus = customers.find(c => c.id === o.customerId);
          const emp = employees.find(e => e.id === o.empId);
          return (
            <div key={o.id} className="bg-white p-4 rounded-xl shadow-sm border border-gray-50 relative group">
              {isAccounting && <button onClick={() => { if(window.confirm('Xóa vĩnh viễn đơn hàng này?')) onDeleteOrder(o.id); }} className="absolute top-3 right-3 text-gray-300 hover:text-red-500"><Trash2 size={14}/></button>}
              <div className="flex justify-between items-start mb-2 pr-6">
                <h3 className="font-bold text-gray-800">{cus?.name}</h3>
                <span className="text-[10px] bg-emerald-50 text-emerald-600 px-2 py-0.5 rounded font-bold">Đã giao</span>
              </div>
              <div className="text-xs text-gray-500 mb-3 border-b border-gray-50 pb-3">
                {formatTime(o.date)} {new Date(o.date).toLocaleDateString('vi-VN')} - HD{o.id.substring(o.id.length-6).toUpperCase()}<br/>
                Tạo bởi {emp?.name || 'Admin'}
                {o.extraExpenseName && o.customerExtraExpense > 0 && <span className="block mt-1 text-blue-600 font-medium">+ {o.extraExpenseName} ({formatCurrency(o.customerExtraExpense)})</span>}
              </div>
              <div className="flex justify-between items-center">
                <span className="text-sm text-gray-600">Tổng cộng</span>
                <div className="text-right">
                  <span className="text-base font-black text-gray-800 block">{formatCurrency(o.amount)} đ</span>
                  <span className="text-[10px] text-orange-500 font-medium">Ghi nhận nợ</span>
                </div>
              </div>
            </div>
          )
        })}
      </div>

      <div className="fixed bottom-[70px] right-4 left-4 pointer-events-none flex justify-center">
         <button onClick={() => setShowAddOrder(true)} className="pointer-events-auto bg-blue-600 text-white rounded-full px-6 py-3 font-bold shadow-lg flex items-center gap-2 hover:bg-blue-700">
            <Plus size={18}/> Bán hàng
         </button>
      </div>

      {showAddOrder && (
         <div className="fixed inset-0 bg-gray-50 z-50 flex flex-col animate-in slide-in-from-bottom">
            <header className="bg-blue-600 text-white p-4 flex items-center justify-between shrink-0 shadow-sm">
               <h2 className="font-bold text-lg">Tạo Đơn Hàng Mới</h2>
               <button onClick={() => setShowAddOrder(false)} className="hover:bg-blue-700 p-1 rounded-full"><X size={24}/></button>
            </header>
            
            <div className="p-4 flex-1 overflow-y-auto">
               <form id="order-form" onSubmit={handleAddSubmit} className="space-y-4 max-w-lg mx-auto pb-6">
                {errorMsg && <div className="bg-red-50 text-red-600 p-3 rounded-lg text-sm mb-4 font-medium border border-red-100">{errorMsg}</div>}
                
                <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                  <label className="block text-xs font-bold text-gray-500 uppercase mb-2">1. Khách hàng *</label>
                  <div className="flex gap-2 items-center">
                    <div className="relative flex-1">
                      <Search size={18} className="absolute left-3 top-3 text-gray-400" />
                      <input type="text" value={searchCus} onChange={e => { setSearchCus(e.target.value); setShowCusDropdown(true); setNewOrder({...newOrder, customerId: ''}); }} onFocus={() => setShowCusDropdown(true)} className="w-full border border-gray-300 p-2.5 pl-10 rounded-xl text-sm outline-none focus:ring-2 focus:ring-blue-500" placeholder="Tìm tên khách hàng..." />
                      {showCusDropdown && (
                        <ul className="absolute z-20 w-full mt-1 bg-white border border-gray-200 rounded-lg shadow-xl max-h-48 overflow-y-auto">
                          {filteredCustomers.map(c => (
                            <li key={c.id} onClick={() => { setNewOrder({...newOrder, customerId: c.id}); setSearchCus(c.name); setShowCusDropdown(false); }} className="p-3 text-sm hover:bg-blue-50 cursor-pointer border-b border-gray-50">
                              <span className="font-semibold block text-gray-800">{c.name}</span><span className="text-[10px] text-gray-500">{c.phone || c.address}</span>
                            </li>
                          ))}
                          {filteredCustomers.length === 0 && <li className="p-3 text-sm text-gray-500 text-center">Không tìm thấy</li>}
                        </ul>
                      )}
                    </div>
                    <button type="button" onClick={() => setShowQuickAddCus(true)} className="bg-emerald-50 text-emerald-600 px-3.5 py-2.5 rounded-xl border border-emerald-200 flex items-center justify-center font-bold hover:bg-emerald-100 transition-colors shrink-0">
                      <Plus size={20} />
                    </button>
                  </div>
                  {newOrder.customerId && (
                     <div className="mt-3 p-2.5 bg-blue-50 border border-blue-100 rounded-lg flex justify-between items-center">
                        <span className="text-sm font-semibold text-blue-800 flex items-center gap-2"><CheckCircle size={16}/> Đã chọn: {customers.find(c => c.id === newOrder.customerId)?.name}</span>
                        <button type="button" onClick={() => {setNewOrder({...newOrder, customerId: ''}); setSearchCus('');}}><X size={16} className="text-blue-500 hover:text-blue-700"/></button>
                     </div>
                  )}
                </div>

                <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                  <div className="flex justify-between items-center mb-3">
                    <label className="block text-xs font-bold text-gray-500 uppercase">2. Sản phẩm</label>
                    <button type="button" onClick={handleAddItem} className="text-xs text-blue-600 font-semibold flex items-center bg-blue-50 px-2.5 py-1.5 rounded-lg hover:bg-blue-100 transition-colors"><Plus size={14} className="mr-1"/> Thêm dòng</button>
                  </div>
                  <div className="space-y-3">
                    {newOrder.items.map((item, index) => (
                      <div key={index} className="relative bg-gray-50 p-3 rounded-xl border border-gray-100">

                        <select value={products.find(p=>p.name === item.description)?.id || ''} onChange={e=>handleItemProductChange(index, e.target.value)} className="w-full border border-gray-200 p-2.5 rounded-lg text-sm outline-none bg-white mb-2 focus:ring-2 focus:ring-blue-500">
                          <option value="">-- Chọn nhanh từ kho --</option>
                          {products.filter(p => !p.isArchived).map(p => <option key={p.id} value={p.id}>{p.name} ({formatCurrency(p.sellingPrice)})</option>)}
                        </select>
                        <input type="text" placeholder="Hoặc nhập tên hàng hóa thủ công..." value={item.description} onChange={e=>handleItemChange(index, 'description', capitalizeFirst(e.target.value))} className="w-full border border-gray-200 p-2.5 rounded-lg text-sm mb-2 outline-none focus:ring-2 focus:ring-blue-500" />

                            <input required type="tel" placeholder="Đơn giá (VNĐ)" value={formatInputCurrency(item.unitPrice)} onChange={e=>handleItemChange(index, 'unitPrice', parseInputCurrency(e.target.value))} className="w-2/3 border border-gray-200 p-2.5 rounded-lg text-sm outline-none focus:ring-2 focus:ring-blue-500" />
                          </div>
                        </div>
                        <div className="text-right text-xs text-blue-600 font-semibold mt-3 pt-3 border-t border-dashed border-gray-200">Tạm tính: {formatCurrency((parseFloat(item.quantity)||0) * (parseFloat(item.unitPrice)||0))} đ</div>
                      </div>
                    ))}
                  </div>
                </div>

                <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100">
                  <label className="block text-xs font-bold text-gray-500 uppercase mb-3">3. Thanh toán & Phụ phí</label>
                  <div className="flex justify-between items-center mb-2"><span className="text-sm text-gray-600">Tổng tiền hàng:</span><span className="font-semibold">{formatCurrency(subTotal)} đ</span></div>
                  <div className="flex justify-between items-center mb-3 pb-3 border-b border-gray-100">
                    <span className="text-sm text-gray-600">Giảm giá / C.Khấu:</span>
                    <input type="tel" value={formatInputCurrency(newOrder.discount)} onChange={e=>setNewOrder({...newOrder, discount: parseInputCurrency(e.target.value)})} className="w-1/2 border border-gray-200 p-2 rounded-lg text-sm text-right outline-none focus:ring-2 focus:ring-blue-500" placeholder="0 ₫" />
                  </div>

                  <div className="mb-4 bg-gray-50 p-3 rounded-xl border border-gray-200">
                    <p className="text-xs font-bold text-gray-700 mb-2">Chi phí khác (Vận chuyển, thùng xốp...)</p>
                    <div className="flex gap-2 mb-2">
                      <input type="text" placeholder="Tên phí (VD: Ship...)" value={newOrder.extraExpenseName} onChange={e=>setNewOrder({...newOrder, extraExpenseName: capitalizeFirst(e.target.value)})} className="flex-1 border border-gray-200 p-2.5 rounded-lg text-sm outline-none focus:ring-2 focus:ring-emerald-500" />
                      <input type="tel" placeholder="Số tiền" value={formatInputCurrency(newOrder.extraExpenseAmount)} onChange={e=>setNewOrder({...newOrder, extraExpenseAmount: parseInputCurrency(e.target.value)})} className="w-1/3 border border-gray-200 p-2.5 rounded-lg text-sm outline-none focus:ring-2 focus:ring-emerald-500" />
                    </div>
                    {parseFloat(newOrder.extraExpenseAmount) > 0 && (
                      <select value={newOrder.extraExpensePayer} onChange={e=>setNewOrder({...newOrder, extraExpensePayer: e.target.value})} className="w-full border border-gray-200 p-2.5 rounded-lg text-sm outline-none bg-white text-emerald-700 font-medium focus:ring-2 focus:ring-emerald-500">

              </div>
            </div>

            {showQuickAddCus && (

                <div className="bg-white rounded-2xl p-5 w-full max-w-sm">
                  <h3 className="font-bold text-lg mb-4 text-gray-800">Thêm Khách Hàng Nhanh</h3>
                  <form onSubmit={handleQuickAddCustomer} className="space-y-4">

                    <input required type="tel" value={newCus.phone} onChange={e=>setNewCus({...newCus, phone: e.target.value})} className="w-full border p-3 rounded-xl outline-none focus:ring-2 focus:ring-emerald-500 text-sm" placeholder="Số điện thoại *" />
                    <input type="text" value={newCus.address} onChange={e=>setNewCus({...newCus, address: toTitleCase(e.target.value)})} className="w-full border p-3 rounded-xl outline-none focus:ring-2 focus:ring-emerald-500 text-sm" placeholder="Địa chỉ giao hàng" />
                    
                    <div className="flex gap-2 pt-2">
                      <button type="button" onClick={()=>setShowQuickAddCus(false)} className="flex-1 bg-gray-100 text-gray-600 py-3.5 rounded-xl font-bold transition-colors">Hủy</button>
                      <button type="submit" className="flex-[2] bg-emerald-500 text-white py-3.5 rounded-xl font-bold transition-colors shadow-md shadow-emerald-500/30">Lưu Khách Hàng</button>
                    </div>
                  </form>
                </div>
              </div>
            )}
         </div>
      )}
    </div>
  );
}

function ProductManagementView({ isAccounting, products, onAddProduct, onEditProduct, onToggleArchiveProduct, onDeleteProduct }) {
  const [showForm, setShowForm] = useState(false);
  const [showArchived, setShowArchived] = useState(false);
  const [actionItem, setActionItem] = useState(null);
  const [editingProd, setEditingProd] = useState(null);
  
  const [prodData, setProdData] = useState({ 
    name: '', category: '', unit: 'Cái', costPrice: '', sellingPrice: '', 
    barcode: '', image: '', discount: '' 
  });

  const [activeCategory, setActiveCategory] = useState('Tất cả');

  const activeProducts = products.filter(p => !!p.isArchived === showArchived);
  
  const categories = useMemo(() => {
    const cats = activeProducts.map(p => p.category).filter(Boolean);
    return ['Tất cả', ...new Set(cats)];
  }, [activeProducts]);

  const displayedProducts = useMemo(() => {
    if (activeCategory === 'Tất cả') return activeProducts;
    return activeProducts.filter(p => p.category === activeCategory);
  }, [activeProducts, activeCategory]);

  const handleOpenEdit = (p) => {
    setEditingProd(p);
    setProdData({ 
      name: p.name, category: p.category, unit: p.unit, 
      costPrice: p.costPrice || '', sellingPrice: p.sellingPrice || '',
      barcode: p.barcode || '', image: p.image || '', discount: p.discount || ''
    });
    setShowForm(true);
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    const pData = { 
      ...prodData, 
      costPrice: parseInt(prodData.costPrice) || 0, 
      sellingPrice: parseInt(prodData.sellingPrice) || 0,
      discount: parseInt(prodData.discount) || 0
    };
    if (editingProd) onEditProduct(editingProd.id, pData); 
    else onAddProduct(pData);
    setShowForm(false);
  };

  const handleImageUpload = (e) => {
    const file = e.target.files[0];
    if (file) {
      if (file.size > 500000) {
        alert("Vui lòng chọn ảnh có kích thước nhỏ hơn 500KB để tối ưu tốc độ.");
        return;
      }
      const reader = new FileReader();
      reader.onloadend = () => setProdData({...prodData, image: reader.result});
      reader.readAsDataURL(file);
    }
  };

  const handleScanBarcode = () => {
    const code = prompt("Mô phỏng máy quét: Vui lòng nhập hoặc dùng súng bắn mã vạch bắn vào đây:", Math.floor(Math.random() * 1000000000000).toString());
    if (code) setProdData({...prodData, barcode: code});
  };

  return (

                </div>
              </div>

      />

}

function CustomerCRMView({ employee, customers, orders, payments, onAddCustomer, employees }) {
  const [showAddCustomer, setShowAddCustomer] = useState(false);
  const activeCustomers = customers.filter(c => !c.isArchived);
  
  const isSales = ['Kinh doanh', 'Bán hàng'].includes(employee?.position);
  const salesEmployees = employees ? employees.filter(e => ['Kinh doanh', 'Bán hàng'].includes(e.position)) : [];
  const [newCus, setNewCus] = useState({ name: '', phone: '', address: '', empId: employee?.id || '' });

  const handleAddCustomerSubmit = (e) => { 
    e.preventDefault(); 
    const finalEmpId = isSales ? employee.id : newCus.empId;
    if (!finalEmpId) return;
    onAddCustomer(finalEmpId, { name: newCus.name, phone: newCus.phone, address: newCus.address }); 
    setShowAddCustomer(false); 
    setNewCus({name:'', phone:'', address:'', empId: employee?.id || ''}); 
  };

  return (
    <div className="space-y-4 animate-in fade-in pb-16">
      <div className="flex border-b border-gray-200">
        <button className="flex-1 py-2 text-sm font-semibold border-b-2 border-emerald-500 text-emerald-600">Khách hàng</button>
        <button className="flex-1 py-2 text-sm font-semibold border-b-2 border-transparent text-gray-500">Nhóm KH</button>
      </div>

      <div className="bg-white px-3 py-2 rounded-lg border border-gray-100 shadow-sm flex items-center gap-2 mb-2">
        <Search size={18} className="text-gray-400" />
        <input type="text" placeholder="Tìm kiếm khách hàng..." className="flex-1 text-sm outline-none" />
      </div>

      <div className="bg-white rounded-xl shadow-sm border border-gray-50 overflow-hidden divide-y divide-gray-50">
        {activeCustomers.map(cus => {
          const initials = (cus.name || '').split(' ').map(n=>n[0]).join('').substring(0,2).toUpperCase();
          return (
            <div key={cus.id} className="flex items-center p-3 gap-3 hover:bg-gray-50 transition cursor-pointer">
              <div className="w-10 h-10 rounded-full bg-emerald-100 text-emerald-700 flex items-center justify-center font-bold text-sm shrink-0">
                {initials}
              </div>
              <div className="flex-1">
                <h3 className="font-semibold text-gray-800 text-sm">{cus.name}</h3>
                <p className="text-xs text-gray-500">{cus.phone || cus.address}</p>
              </div>
              <div className="text-emerald-500"><Banknote size={16} /></div>
            </div>
          )
        })}
      </div>

      <div className="fixed bottom-[70px] right-4 pointer-events-none flex justify-end">
         <button onClick={() => setShowAddCustomer(true)} className="pointer-events-auto bg-blue-600 text-white rounded-full px-5 py-3 font-bold shadow-lg flex items-center gap-2 hover:bg-blue-700">
            <Plus size={18}/> Tạo khách hàng
         </button>
      </div>

      {showAddCustomer && (
        <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl p-5 w-full max-w-sm animate-in zoom-in-95">
            <h3 className="font-bold text-lg mb-4">Thêm Khách Hàng</h3>
            <form onSubmit={handleAddCustomerSubmit} className="space-y-4">
              <input required type="text" value={newCus.name} onChange={e=>setNewCus({...newCus, name: toTitleCase(e.target.value)})} className="w-full border p-3 rounded-xl outline-none focus:ring-2 focus:ring-emerald-500 text-sm" placeholder="Tên khách hàng/Công ty" />
              <input required type="tel" value={newCus.phone} onChange={e=>setNewCus({...newCus, phone: e.target.value})} className="w-full border p-3 rounded-xl outline-none focus:ring-2 focus:ring-emerald-500 text-sm" placeholder="Số điện thoại" />
              <input type="text" value={newCus.address} onChange={e=>setNewCus({...newCus, address: toTitleCase(e.target.value)})} className="w-full border p-3 rounded-xl outline-none focus:ring-2 focus:ring-emerald-500 text-sm" placeholder="Địa chỉ giao hàng" />
              {!isSales && (
                <div>
                  <label className="block text-xs font-bold text-gray-500 mb-1">Nhân viên phụ trách</label>
                  <select required value={newCus.empId} onChange={e=>setNewCus({...newCus, empId: e.target.value})} className="w-full border p-3 rounded-xl outline-none bg-white text-sm">
                    <option value="">-- Chọn Sales phụ trách --</option>
                    {salesEmployees.map(e => <option key={e.id} value={e.id}>{e.name}</option>)}
                  </select>
                </div>
              )}
              <button type="submit" className="w-full bg-emerald-500 text-white py-3 rounded-xl font-bold mt-2">Lưu Khách Hàng</button>
              <button type="button" onClick={()=>setShowAddCustomer(false)} className="w-full bg-gray-100 text-gray-600 py-3 rounded-xl font-bold">Hủy</button>
            </form>
          </div>
        </div>
      )}
    </div>
  );
}

function EmployeeView({ employees, onAddEmployee, onEditEmployee, isSuperAdmin }) {
  const [showForm, setShowForm] = useState(false);
  const [editingEmp, setEditingEmp] = useState(null);
  const [empData, setEmpData] = useState({ name: '', phone: '', position: 'Kinh doanh', startDate: getTodayString(), probationDuration: '0', probationUnit: 'days', probationRate: '100', basicSalary: '', supportSalary: '', experienceSalary: '', commissionRate: '', targetRevenue: '', overtimeRate: '' });

  const handleOpenEdit = (emp) => {
    setEditingEmp(emp);
    setEmpData({ 
      name: emp.name, phone: emp.phone, position: emp.position, 
      startDate: emp.startDate || getTodayString(), 
      probationDuration: emp.probationDuration || 0,
      probationUnit: emp.probationUnit || 'days',
      probationRate: emp.probationRate !== undefined ? emp.probationRate : 100,
      basicSalary: emp.basicSalary, supportSalary: emp.supportSalary || 0, experienceSalary: emp.experienceSalary || 0, commissionRate: emp.commissionRate || 0, targetRevenue: emp.targetRevenue || 0, overtimeRate: emp.overtimeRate || 0 
    });

  };

    e.preventDefault();
    const pData = { 
      ...empData, 
      probationDuration: parseInt(empData.probationDuration)||0,
      probationRate: parseInt(empData.probationRate)||100,
      basicSalary: parseInt(empData.basicSalary)||0,

        ))}
      </div>

                    <input type="tel" placeholder="Chỉ Tiêu Doanh Số" value={formatInputCurrency(empData.targetRevenue)} onChange={e=>setEmpData({...empData, targetRevenue: parseInputCurrency(e.target.value)})} className="w-full border border-emerald-200 p-2.5 rounded-lg text-sm"/>

                </div>
              )}
              
              <div className="flex gap-2 mt-4 pt-2">
                <button type="button" onClick={()=>setShowForm(false)} className="flex-1 bg-gray-100 py-3 rounded-xl text-gray-700 font-bold">Hủy</button>

              </div>
            </form>
          </div>
        </div>
      )}
    </div>
  );
}

function SalaryView({ employees, attendance, financials, performance, customers, orders, payments, holidays, onAddFinancial, onUpdatePerformance, advanceRequests, onApproveAdvance, onRejectAdvance }) {
  const [selectedEmp, setSelectedEmp] = useState(null);
  const [showModal, setShowModal] = useState(false);
  const [modalType, setModalType] = useState('bonus'); 
  const [amountInput, setAmountInput] = useState('');
  const [reasonInput, setReasonInput] = useState('');
  const [overtimeInput, setOvertimeInput] = useState('');
  const [expandedEmp, setExpandedEmp] = useState(null);

  const aggregateData = useMemo(() => {
    let totalSalary = 0; let totalDays = 0;
    employees.forEach(emp => {
      const details = calculateSalaryDetails(emp.id, employees, attendance, financials, performance, customers, orders, payments, holidays);
      if (details) { totalSalary += details.netSalary; totalDays += details.workDays; }
    });
    return { totalSalary, totalDays, avgDaily: totalDays > 0 ? totalSalary / totalDays : 0 };
  }, [employees, attendance, financials, performance, customers, orders, payments, holidays]);

  const handleSubmit = (e) => {
    e.preventDefault();
    if (modalType === 'performance') onUpdatePerformance(selectedEmp.id, overtimeInput);
    else { 
      const val = parseInputCurrency(amountInput);
      if (!val || isNaN(val)) return; 
      onAddFinancial(selectedEmp.id, modalType, val, capitalizeFirst(reasonInput)); 
    }
    setShowModal(false);
  };

  const pendingAdvances = advanceRequests ? advanceRequests.filter(r => r.status === 'pending') : [];

  return (
    <div className="space-y-4 animate-in fade-in pb-16">
      {pendingAdvances.length > 0 && (
        <div className="bg-orange-50 border border-orange-200 p-4 rounded-2xl shadow-sm mb-6">
          <h2 className="font-bold text-orange-800 mb-3 flex items-center text-sm"><AlertCircle size={16} className="mr-1"/> Yêu Cầu Tạm Ứng</h2>
          <div className="space-y-2">
            {pendingAdvances.map(req => {
              const e = employees.find(emp => emp.id === req.empId);
              return (
                <div key={req.id} className="bg-white p-3 rounded-xl border border-orange-100 flex flex-col shadow-sm">
                  <div className="flex justify-between items-center mb-2">
                    <span className="font-bold text-gray-800 text-sm">{e?.name}</span>
                    <span className="text-orange-600 font-bold">{formatCurrency(req.amount)} đ</span>
                  </div>
                  <p className="text-xs text-gray-500 mb-3">Lý do: {req.reason}</p>
                  <div className="flex gap-2">
                    <button onClick={() => onRejectAdvance(req.id)} className="flex-1 bg-gray-100 text-gray-700 py-2 rounded-lg text-xs font-bold">Từ chối</button>
                    <button onClick={() => onApproveAdvance(req.id)} className="flex-1 bg-emerald-500 text-white py-2 rounded-lg text-xs font-bold">Phê duyệt</button>
                  </div>
                </div>
              );
            })}
          </div>
        </div>
      )}

      <div className="bg-gradient-to-br from-emerald-500 to-emerald-700 text-white p-5 rounded-2xl shadow-md mb-6">
        <p className="text-emerald-100 text-xs font-bold uppercase tracking-wider mb-1">Tổng quỹ lương dự kiến</p>
        <h3 className="text-3xl font-black mb-4">{formatCurrency(aggregateData.totalSalary)} đ</h3>
        <div className="grid grid-cols-2 gap-4 border-t border-emerald-400/50 pt-3">
          <div><p className="text-emerald-100 text-[10px] uppercase mb-1">Tổng ngày công</p><p className="font-bold text-sm">{aggregateData.totalDays}</p></div>
          <div><p className="text-emerald-100 text-[10px] uppercase mb-1">Trung bình/ngày</p><p className="font-bold text-sm">{formatCurrency(aggregateData.avgDaily)}</p></div>
        </div>
      </div>
      
      <div className="space-y-3">
        {employees.map(emp => {
          const details = calculateSalaryDetails(emp.id, employees, attendance, financials, performance, customers, orders, payments, holidays);
          const isExpanded = expandedEmp === emp.id;
          const isSales = ['Kinh doanh', 'Bán hàng'].includes(emp.position);
          if(!details) return null;
          return (
            <div key={emp.id} className="bg-white rounded-xl shadow-sm border border-gray-50 overflow-hidden transition-all">
              <div className="p-4 cursor-pointer" onClick={() => setExpandedEmp(isExpanded ? null : emp.id)}>
                <div className="flex justify-between items-center">
                  <div>
                    <h3 className="font-bold text-gray-800 text-sm flex items-center">{emp.name}</h3>
                    <p className="text-[10px] text-gray-500 mt-1 flex gap-2">
                      <span className="bg-gray-100 px-1.5 py-0.5 rounded">Công: {details.workDays}</span> 
                      <span className="bg-gray-100 px-1.5 py-0.5 rounded">TC: {details.perf.overtime}h</span>
                    </p>
                  </div>
                  <div className="text-right flex items-center gap-2">
                    <span className={`font-bold ${details.netSalary < 0 ? 'text-red-500' : 'text-emerald-600'}`}>{formatCurrency(details.netSalary)} đ</span>
                    {isExpanded ? <ChevronUp size={16} className="text-gray-400"/> : <ChevronDown size={16} className="text-gray-400"/>}
                  </div>
                </div>
              </div>

              {isExpanded && (
                <div className="px-4 pb-4 border-t border-gray-50 pt-3 bg-gray-50/50">
                  <div className="space-y-3 text-xs">
                    <div className="grid grid-cols-2 gap-2 text-gray-600">
                      <div className="flex justify-between"><span>Lương CB:</span> <strong>{formatCurrency(details.baseSalaryCalc)}</strong></div>
                      <div className="flex justify-between"><span>Hỗ trợ:</span> <strong>{formatCurrency(details.supportSalary + details.experienceSalary)}</strong></div>
                      {isSales && <div className="flex justify-between text-emerald-600"><span>Hoa hồng:</span> <strong>+{formatCurrency(details.commission)}</strong></div>}
                      <div className="flex justify-between text-emerald-600"><span>Tăng ca:</span> <strong>+{formatCurrency(details.overtimePay)}</strong></div>
                      <div className="flex justify-between text-emerald-600"><span>Thưởng:</span> <strong>+{formatCurrency(details.totalBonus)}</strong></div>
                      <div className="flex justify-between text-red-500"><span>Phạt/Ứng:</span> <strong>-{formatCurrency(details.totalPenalty + details.totalAdvance)}</strong></div>
                    </div>
                  </div>
                  <div className="flex space-x-2 mt-4 pt-3 border-t border-gray-100">
                    <button onClick={(e) => { e.stopPropagation(); setSelectedEmp(emp); setModalType('performance'); setOvertimeInput(details.perf.overtime.toString()); setShowModal(true); }} className="flex-1 bg-white text-gray-700 py-1.5 rounded border border-gray-200 text-[10px] font-bold flex justify-center items-center gap-1"><Clock size={12}/> Giờ Tăng Ca</button>
                    <button onClick={(e) => { e.stopPropagation(); setSelectedEmp(emp); setModalType('bonus'); setAmountInput(''); setReasonInput(''); setShowModal(true); }} className="flex-1 bg-white text-emerald-600 py-1.5 rounded border border-emerald-200 text-[10px] font-bold flex justify-center items-center gap-1"><Plus size={12}/> Thưởng</button>
                    <button onClick={(e) => { e.stopPropagation(); setSelectedEmp(emp); setModalType('penalty'); setAmountInput(''); setReasonInput(''); setShowModal(true); }} className="flex-1 bg-white text-red-500 py-1.5 rounded border border-red-200 text-[10px] font-bold flex justify-center items-center gap-1"><MinusCircle size={12}/> Phạt</button>
                  </div>
                </div>
              )}
            </div>
          );
        })}
      </div>

      {showModal && (
        <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
          <div className="bg-white rounded-2xl p-5 w-full max-w-sm">
            <div className="flex justify-between items-center mb-4"><h3 className="font-bold text-lg text-gray-800">{modalType === 'performance' ? 'Cập nhật Giờ làm thêm' : modalType === 'bonus' ? 'Thêm Tiền Thưởng' : 'Trừ Tiền Phạt'}</h3><button onClick={() => setShowModal(false)} className="text-gray-400"><X /></button></div>
            <form onSubmit={handleSubmit} className="space-y-4">
              {modalType === 'performance' ? (
                <div><label className="block text-xs font-semibold text-gray-700 mb-1">Số giờ tăng ca</label><input type="number" step="0.5" value={overtimeInput} onChange={(e) => setOvertimeInput(e.target.value)} className="w-full border border-gray-300 rounded-lg p-2.5 outline-none focus:ring-2 focus:ring-blue-500" /></div>
              ) : (
                <><div className="mb-2"><label className="block text-xs font-semibold text-gray-700 mb-1">Số tiền (VNĐ)</label><input type="tel" required value={formatInputCurrency(amountInput)} onChange={(e) => setAmountInput(parseInputCurrency(e.target.value))} className="w-full border border-gray-300 rounded-lg p-2.5 outline-none focus:ring-2 focus:ring-emerald-500" /></div>
                <div><label className="block text-xs font-semibold text-gray-700 mb-1">Lý do</label><input type="text" required value={reasonInput} onChange={(e) => setReasonInput(capitalizeFirst(e.target.value))} className="w-full border border-gray-300 rounded-lg p-2.5 outline-none focus:ring-2 focus:ring-emerald-500" /></div></>
              )}
              <button type="submit" className="w-full bg-emerald-500 hover:bg-emerald-600 transition-colors text-white py-3 rounded-lg font-semibold mt-4">Xác nhận</button>
            </form>
          </div>
        </div>
      )}
    </div>
  );
}

function DebtManagementView({ isAccounting, isDriver, employee, customers, orders, payments, employees, onAddPayment }) {
  const [selectedCustomer, setSelectedCustomer] = useState(null);
  const [showPaymentModal, setShowPaymentModal] = useState(false);
  const [newPayment, setNewPayment] = useState({ amount: '', note: '', date: getTodayString() });

  const totalCompanyDebt = customers.reduce((sum, cus) => sum + getCustomerDebt(cus.id, orders, payments), 0);

  const handlePaymentSubmit = (e) => {
    e.preventDefault();
    if (!selectedCustomer) return;
    onAddPayment({ customerId: selectedCustomer.id, amount: parseInt(parseInputCurrency(newPayment.amount)) || 0, note: newPayment.note || (isDriver ? 'Tài xế thu tiền mặt' : 'Thu nợ'), date: newPayment.date });
    setShowPaymentModal(false); setNewPayment({ amount: '', note: '', date: getTodayString() });
  };

  if (selectedCustomer) {
    const currentDebt = getCustomerDebt(selectedCustomer.id, orders, payments);
    return (
      <div className="space-y-4 animate-in fade-in pb-16">
        <button onClick={() => setSelectedCustomer(null)} className="text-gray-500 flex items-center text-sm font-semibold mb-2"><ChevronLeft size={18} className="mr-1" /> Trở về danh sách</button>
        <div className="bg-white p-5 rounded-2xl shadow-sm border border-gray-50">
          <h2 className="font-bold text-xl text-gray-800 mb-4">{selectedCustomer.name}</h2>
          <div className="bg-red-50 border border-red-100 rounded-xl p-4 flex justify-between items-center mb-6">
            <span className="text-red-800 font-medium">Khách Đang Nợ:</span>
            <span className="text-2xl font-black text-red-600">{formatCurrency(currentDebt)} đ</span>
          </div>
          
          <button onClick={() => setShowPaymentModal(true)} className="w-full py-4 rounded-xl font-bold text-white shadow-lg flex justify-center items-center gap-2 bg-blue-600 hover:bg-blue-700">
            {isDriver ? <Banknote size={20}/> : <Wallet size={20}/>} {isDriver ? 'Báo Tài Xế Đã Thu Tiền' : 'Lập Phiếu Thu Nợ'}
          </button>
        </div>

        {showPaymentModal && (
          <div className="fixed inset-0 bg-black/60 flex items-center justify-center z-50 p-4">
            <div className="bg-white rounded-2xl p-5 w-full max-w-sm animate-in zoom-in-95">
              <h3 className="font-bold text-lg mb-4">{isDriver ? 'Báo Thu Tiền' : 'Ghi nhận Thanh toán'}</h3>
              <form onSubmit={handlePaymentSubmit} className="space-y-3">
                <input required type="tel" value={formatInputCurrency(newPayment.amount)} onChange={e=>setNewPayment({...newPayment, amount: parseInputCurrency(e.target.value)})} className="w-full border p-3 rounded-xl text-lg font-bold text-emerald-600 outline-none" placeholder="Số tiền (VNĐ)" />
                <input required type="text" value={newPayment.note} onChange={e=>setNewPayment({...newPayment, note: capitalizeFirst(e.target.value)})} className="w-full border p-3 rounded-xl text-sm outline-none" placeholder="Ghi chú" />
                <div className="flex gap-2 mt-4">
                  <button type="button" onClick={() => setShowPaymentModal(false)} className="flex-1 bg-gray-100 py-3 rounded-xl text-gray-700 font-bold">Hủy</button>
                  <button type="submit" className="flex-1 bg-emerald-500 text-white py-3 rounded-xl font-bold">Xác nhận thu</button>
                </div>
              </form>
            </div>
          </div>
        )}
      </div>
    );
  }

  return (
    <div className="space-y-4 animate-in fade-in pb-16">
      {isAccounting && (
        <div className="bg-gradient-to-r from-red-500 to-orange-500 p-5 rounded-2xl shadow-md text-white mb-6">
          <p className="text-red-100 text-xs font-bold uppercase tracking-wider mb-1">Tổng Công Nợ Thị Trường</p>
          <h3 className="text-3xl font-black">{formatCurrency(totalCompanyDebt)} đ</h3>
        </div>
      )}
      
      <div className="bg-white px-3 py-2 rounded-lg border border-gray-100 shadow-sm flex items-center gap-2 mb-4">
        <Search size={18} className="text-gray-400" />
        <input type="text" placeholder="Tìm kiếm khách nợ..." className="flex-1 text-sm outline-none" />
      </div>

      <div className="space-y-3">
        {customers.map(cus => {
          const debt = getCustomerDebt(cus.id, orders, payments);
          if (debt <= 0) return null; 
          return (
            <div key={cus.id} onClick={() => setSelectedCustomer(cus)} className="bg-white p-4 rounded-xl shadow-sm border border-gray-50 flex justify-between items-center cursor-pointer hover:bg-gray-50 transition">
              <div>
                <h3 className="font-bold text-gray-800 text-sm">{cus.name}</h3>
                <span className="text-[10px] text-gray-500 flex items-center mt-1"><MapPin size={10} className="mr-1"/> {cus.address}</span>
              </div>
              <div className="text-right">
                <span className="block font-bold text-red-500">{formatCurrency(debt)} đ</span>
                <span className="text-[10px] text-gray-400">Cần thu</span>
              </div>
            </div>
          )
        })}
      </div>
    </div>
  );
}

function LoginRegisterView({ onLogin, onRegister }) {
  const [isLogin, setIsLogin] = useState(true);
  const [loginPhone, setLoginPhone] = useState('');
  const [loginError, setLoginError] = useState('');
  const [regCompanyName, setRegCompanyName] = useState('');
  const [regPhone, setRegPhone] = useState('');
  const [regError, setRegError] = useState('');
  const [isRegistering, setIsRegistering] = useState(false);

  const handleLoginSubmit = (e) => {
    e.preventDefault();
    if (!loginPhone.trim()) { setLoginError('Vui lòng nhập số điện thoại'); return; }
    if (!onLogin(loginPhone)) setLoginError('Số điện thoại không đúng hoặc công ty chưa đăng ký.');
  };

  const handleRegisterSubmit = async (e) => {
    e.preventDefault();
    if (!regCompanyName.trim() || !regPhone.trim()) { setRegError('Vui lòng điền đủ thông tin'); return; }
    setIsRegistering(true);
    const result = await onRegister(toTitleCase(regCompanyName), regPhone);
    if (!result.success) setRegError(result.message);
    setIsRegistering(false);
  };

  return (
    <div className="flex flex-col h-screen bg-[#f4f6f8] max-w-md mx-auto relative overflow-hidden items-center justify-center p-6 font-sans">
      <div className="bg-white w-full p-8 rounded-3xl shadow-[0_8px_30px_rgb(0,0,0,0.04)] border border-gray-50 animate-in fade-in zoom-in-95 duration-300">
        
        {/* --- KHU VỰC LOGO & THƯƠNG HIỆU --- */}
        <div className="flex flex-col items-center justify-center mb-8 space-y-3">
          <div className="w-24 h-24 bg-white rounded-2xl flex items-center justify-center shadow-[0_4px_20px_rgba(0,0,0,0.08)] overflow-hidden border border-gray-50 p-2">
            <img 
              src="https://api.dicebear.com/7.x/shapes/svg?seed=HDManager&backgroundColor=10b981" 
              alt="App Logo" 
              className="w-full h-full object-contain rounded-xl" 
            />
          </div>
          <h1 className="text-2xl font-extrabold text-transparent bg-clip-text bg-gradient-to-r from-emerald-600 to-teal-500 tracking-tight uppercase">
            HD MANAGER
          </h1>
          <p className="text-[10px] text-gray-400 font-bold uppercase tracking-[0.2em]">Nền tảng quản lý số</p>
        </div>
        
        <div className="flex bg-gray-50 p-1.5 rounded-xl mb-8">
          <button onClick={() => {setIsLogin(true); setRegError(''); setLoginError('');}} className={`flex-1 py-2.5 text-sm font-bold rounded-lg transition-all duration-200 ${isLogin ? 'bg-white shadow-sm text-emerald-600' : 'text-gray-400 hover:text-gray-600'}`}>Đăng nhập</button>
          <button onClick={() => {setIsLogin(false); setRegError(''); setLoginError('');}} className={`flex-1 py-2.5 text-sm font-bold rounded-lg transition-all duration-200 ${!isLogin ? 'bg-white shadow-sm text-emerald-600' : 'text-gray-400 hover:text-gray-600'}`}>Tạo Công ty</button>
        </div>

        {isLogin ? (
          <div>
            <h2 className="text-xl font-bold text-center text-gray-800 mb-2">Đăng Nhập</h2>
            <p className="text-center text-xs text-gray-500 mb-6 font-medium">Dành cho chủ doanh nghiệp & nhân viên</p>
            {loginError && <div className="mb-4 text-xs text-red-500 bg-red-50/50 p-3 rounded-xl text-center font-semibold">{loginError}</div>}
            <form onSubmit={handleLoginSubmit} className="space-y-4">
              <input type="tel" value={loginPhone} onChange={(e) => { setLoginPhone(e.target.value); setLoginError(''); }} className="w-full border-2 border-gray-100 rounded-2xl p-4 outline-none focus:border-emerald-500 text-sm font-medium transition-colors" placeholder="Nhập số điện thoại..." />
              <button type="submit" className="w-full bg-emerald-500 hover:bg-emerald-600 text-white py-4 rounded-2xl font-bold shadow-lg shadow-emerald-500/30 transition-all active:scale-[0.98]">Vào Ứng Dụng</button>
            </form>
          </div>
        ) : (
          <div>
            <h2 className="text-xl font-bold text-center text-gray-800 mb-2">Tạo Mới</h2>
            <p className="text-center text-xs text-gray-500 mb-6 font-medium">Tạo vùng dữ liệu Đám mây riêng</p>
            {regError && <div className="mb-4 text-xs text-red-500 bg-red-50/50 p-3 rounded-xl text-center font-semibold">{regError}</div>}
            <form onSubmit={handleRegisterSubmit} className="space-y-4">
              <input type="text" value={regCompanyName} onChange={(e) => { setRegCompanyName(toTitleCase(e.target.value)); setRegError(''); }} className="w-full border-2 border-gray-100 rounded-2xl p-4 outline-none focus:border-emerald-500 text-sm font-medium transition-colors" placeholder="Tên Doanh Nghiệp (VD: Cty HD)" />
              <input type="tel" value={regPhone} onChange={(e) => { setRegPhone(e.target.value); setRegError(''); }} className="w-full border-2 border-gray-100 rounded-2xl p-4 outline-none focus:border-emerald-500 text-sm font-medium transition-colors" placeholder="SĐT Chủ Doanh Nghiệp" />
              <button type="submit" disabled={isRegistering} className="w-full bg-orange-500 hover:bg-orange-600 text-white py-4 rounded-2xl font-bold shadow-lg shadow-orange-500/30 transition-all active:scale-[0.98] mt-2">

    </div>
  );
}
